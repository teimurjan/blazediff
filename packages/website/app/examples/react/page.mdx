import { OnionSkinMode } from '@blazediff/react';
import { useState } from 'react';

function OnionSkinExample() {
  const [opacity, setOpacity] = useState(50);

  return (
    <div className="max-w-4xl mx-auto">
      <OnionSkinMode
        src1="/layers/base.png"
        src2="/layers/overlay.png"
        opacity={opacity}
        sliderLabelText="Blend: "
        className="rounded-lg overflow-hidden shadow-lg"
        containerClassName="bg-white"
        sliderClassName="accent-purple-500"
        sliderLabelClassName="font-medium text-gray-700"
        onOpacityChange={setOpacity}
      />

      <div className="mt-4 text-center space-x-2">
        <span className="text-gray-600">Quick controls:</span>
        {[25, 50, 75].map(value => (
          <button
            key={value}
            onClick={() => setOpacity(value)}
            className={`px-3 py-1 rounded ${
              opacity === value
                ? 'bg-purple-500 text-white'
                : 'bg-gray-200 text-gray-700 hover:bg-gray-300'
            }`}
          >
            {value}%
          </button>
        ))}
      </div>
    </div>
  );
}

import { DifferenceMode } from '@blazediff/react';
import { useState } from 'react';

function DifferenceExample() {
  const [diffResult, setDiffResult] = useState(null);
  const [threshold, setThreshold] = useState(0.1);
  const [loading, setLoading] = useState(false);

  const handleDiffComplete = (result) => {
    setDiffResult(result);
    setLoading(false);
  };

  const handleDiffError = (error) => {
    console.error('Diff calculation failed:', error);
    setLoading(false);
  };

  return (
    <div className="max-w-4xl mx-auto space-y-4">
      <div className="bg-gray-50 p-4 rounded-lg">
        <label className="block text-sm font-medium mb-2">
          Detection Threshold: {(threshold * 100).toFixed(0)}%
        </label>
        <input
          type="range"
          min="0"
          max="100"
          value={threshold * 100}
          onChange={(e) => {
            setThreshold(e.target.value / 100);
            setLoading(true);
          }}
          className="w-full"
        />
      </div>

      <DifferenceMode
        src1="/tests/expected.png"
        src2="/tests/actual.png"
        threshold={threshold}
        includeAA={false}
        alpha={0.2}
        className="border border-gray-200 rounded-lg"
        containerClassName="bg-white"
        canvasClassName="max-w-full h-auto"
        onDiffComplete={handleDiffComplete}
        onDiffError={handleDiffError}
      />

      {diffResult && (
        <div className="bg-gray-50 p-4 rounded-lg">
          <h3 className="font-semibold mb-2">Analysis Results:</h3>
          <div className="grid grid-cols-2 md:grid-cols-4 gap-4 text-sm">
            <div>
              <span className="font-medium">Different pixels:</span>
              <br />
              <span className="text-lg">{diffResult.diffCount.toLocaleString()}</span>
            </div>
            <div>
              <span className="font-medium">Total pixels:</span>
              <br />
              <span className="text-lg">{diffResult.totalPixels.toLocaleString()}</span>
            </div>
            <div>
              <span className="font-medium">Percentage:</span>
              <br />
              <span className="text-lg">{diffResult.percentage.toFixed(4)}%</span>
            </div>
            <div>
              <span className="font-medium">Status:</span>
              <br />
              <span className={`inline-block px-2 py-1 rounded text-xs font-medium ${
                diffResult.diffCount === 0
                  ? 'bg-green-100 text-green-800'
                  : 'bg-red-100 text-red-800'
              }`}>
                {diffResult.diffCount === 0 ? 'Identical' : 'Different'}
              </span>
            </div>
          </div>
        </div>
      )}
    </div>
  );
}
